import random
def névelo(szó):
    maganhangzok = 'aáeéiíoóöőuúüű'
    if szó[0].lower() in maganhangzok:
        return 'az'
    else:return 'a'

def jelzo():
    return random.choice(['könnyed', 'nagy', 'piros'])
print('három főnevet adj meeg!')
for szám in range(1,4):
    főnév = input(str(szám) + '. főnév: ')
    print(névelo(főnév).capitalize(), ' ', főnév, ' ', jelzo(), '.', sep='')
